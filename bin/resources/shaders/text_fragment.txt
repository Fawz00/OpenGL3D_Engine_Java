#version 330 core

precision highp float;

uniform sampler2D TEXTURE;

in vec2 uv;
in vec4 color;

out vec4 FragColor;

void main() {
	vec4 col = vec4(0.0);
	if( texture(TEXTURE, uv).r >= 0.2 && texture(TEXTURE, uv).r < 0.35) col = vec4(color.rgb, color.a/2.);
	if( texture(TEXTURE, uv).r >= 0.35) col = color;
	FragColor = col;
}
